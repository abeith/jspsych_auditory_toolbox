Auditory Experiments

Audio Recorder

const rec = {
    type: PCMRecord,
    stimulus: 'Blue'
};

const play = {
    type: PCMPlay,
    pcmData: () => jsPsych.data.get().last(1).trials[0]
};

const rec_play = {
    timeline: [rec, play],
    loop_function: (data) => {
        let retry = data.values()[1].retry;
        data.values()[0].mic_signal = (retry) ? "delete" : "save";
        return retry;
    }
};
